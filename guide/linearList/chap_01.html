链表（线性表的链式表示） #

  • 大话时刻：链表是通过指针来实现。就是指来指去。有单向链表、双向链表。用针线，把一块块布片连起来，布头串一下，从布尾穿出去，反复如此，再去串下一个布片。

1.2.1 头部插入 #

c
#include <stdio.h>
#include <stdlib.h>

typedef int ElemType;
typedef struct LNode
{
  ElemType data; // 数据域
  struct LNode *next; // 指针域
  
}LNode, *LinkList;

// LNode*是结构体指针，和LinkList是等价的
void head_insert_list(LNode* &L) {
  L = (LNode*) malloc(sizeof(LNode)); // 申请头结点空间，头指针指向头结点
  L -> next = NULL; // 初始化头结点的next指针为NULL

  ElemType x;
  scanf("%d", &x);
  
  // 定义新节点指针
  LNode* P;

  while (x != 9999)
  {
    P = (LinkList)malloc(sizeof(LNode)); // 申请新节点空间， 新节点指针指向新节点
    P -> data = x; // 为新节点的数据域赋值
    P -> next = L -> next; // 新节点的next指针指向原来头结点的next指针
    L -> next = P; // 头结点的next指针指向新节点的指针
    scanf("%d", &x);

  }
  

}

void PrintLinkList(LNode* L) {
  L = L -> next;
  while (L != NULL)
  {
    printf("%3d", L->data);
    L =  L -> next;
  }

  printf("\n");
  
}
/*
1. 定义头指针
2. 申请头结点 头指针指向头结点 初始化头结点next指针为NULL
3. scanf 输入数据 为数据申请结点空间 头结点next指针指向新节点
4. 新节点的next指向头结点的next指针， 新节点data赋值
*/
int main(){
  LNode* L; // 链表头指针 结构体指针类型
  head_insert_list(L);
  PrintLinkList(L);
  return 0;
}

1.2.2 尾部插入 #

C
#include <stdio.h>
#include <stdlib.h>

typedef int ElemType;
typedef struct LNode
{
  ElemType data; // 数据域
  struct LNode *next; // 指针域
  
}LNode, *LinkList;


void PrintLinkList(LNode* L) {
  L = L -> next;
  while (L != NULL)
  {
    printf("%3d", L->data);
    L =  L -> next;
  }

  printf("\n");
  
}

/*
尾插法
1. 定义头指针
2. 申请头结点，头指针指向头结点 头结点的next初始化为NULL
3. 定义一个尾结点指针 用来指向最后一个结点 其初始化指向头结点
4. scanf输入数据 申请结点空间 尾结点的next指向新节点 新节点data赋值 尾结点重置为新节点
5. 尾结点的next为NULL
*/

void tail_insert_list(LinkList &L) {
  L = (LinkList)malloc(sizeof(LNode)); // 申请头结点 头指针指向头结点
  L -> next = NULL; // 初始化头结点的Next指针为NULL
  LinkList tail = L, P;// 定义结构体指针类型 尾结点指向头部结点，新节点
  ElemType x;

  scanf("%d", &x);

  while (x != 9999)
  {
    P = (LinkList)(malloc(sizeof(LNode)));
    P -> data = x;
    tail -> next = P;
    tail = P;
    scanf("%d", &x);
  }
  tail -> next = NULL;
}
int main(){
  LNode* L; // 链表头指针 结构体指针类型
  tail_insert_list(L); // 尾部插入
  PrintLinkList(L);
  return 0;
}

1.2.3 查找：位置查找、按值查找 #

c
#include <stdio.h>
#include <stdlib.h>

typedef int ElemType;
typedef struct LNode
{
  ElemType data; // 数据域
  struct LNode *next; // 指针域
  
}LNode, *LinkList;


void PrintLinkList(LNode* L) {
  L = L -> next;
  while (L != NULL)
  {
    printf("%3d    %x\n", L->data, L -> next);
    L =  L -> next;
  }

  printf("\n");
  
}

/*
尾插法
1. 定义头指针
2. 申请头结点，头指针指向头结点
3. 定义一个尾结点指针 用来指向最后一个结点
4. scanf输入数据 申请结点空间 头结点next指针指向新结点next指针 保存尾结点指针
*/

void tail_insert_list(LinkList &L) {
  L = (LinkList)malloc(sizeof(LNode));
  L -> next = NULL;
  LinkList tail = L, P;
  ElemType x;

  scanf("%d", &x);


  while (x != 9999)
  {
    P = (LinkList)(malloc(sizeof(LNode)));
    P -> data = x;
    tail -> next = P;
    tail = P;
    scanf("%d", &x);
  }
  tail -> next = NULL;
  
}
/*
查找：按位置查找
1. 做边界判断  查找位置不能小于一 如果是0就返回头指针 
2. 做循环 依次赋值出 当前结点的指针
*/
LNode* GetElement(LinkList L, int i) {
  int j = 1;
  if(i == 0) return L;
  if(i < 1) return NULL;
  LinkList P;
  P = L -> next;
  while (j < i)
  {
    P = P -> next;
    j++;
  }
  return P;
}

/*
查找： 值查找

*/
LinkList LocateElement(LinkList L, ElemType el) {
  LinkList P;
  P = L -> next;
  while (P != NULL && P->data != el) {
    P = P -> next;
  }
  return P;
  
}
int main(){
  LNode* L; // 链表头指针 结构体指针类型
  tail_insert_list(L); // 尾部插入
  PrintLinkList(L);
    
  // LinkList res = GetElement(L, 0);
  LinkList res = LocateElement(L, 22);
  if(res) {
    printf("%d  %x\n", res -> data, res -> next);
  }else {
    printf("NULL");
  }
  return 0;
}

1.2.4 在位置i插入元素 #

C
#include <stdio.h>
#include <stdlib.h>

typedef int ElemType;
typedef struct LNode
{
  ElemType data; // 数据域
  struct LNode *next; // 指针域
  
}LNode, *LinkList;


void PrintLinkList(LNode* L) {
  LinkList add = L->next;
  L = L -> next;
  while (L != NULL)
  {
    printf("%3d    %x\n", L->data, add);
    add = L->next;
    L =  L -> next;
  }

  printf("\n");
  
}


/*
尾插法
1. 定义头指针
2. 申请头结点，头指针指向头结点
3. 定义一个尾结点指针 用来指向最后一个结点
4. scanf输入数据 申请结点空间 头结点next指针指向新结点next指针 保存尾结点指针
*/

void tail_insert_list(LinkList &L) {
  L = (LinkList)malloc(sizeof(LNode));
  L -> next = NULL;
  LinkList tail = L, P;
  ElemType x;

  scanf("%d", &x);


  while (x != 9999)
  {
    P = (LinkList)(malloc(sizeof(LNode)));
    P -> data = x;
    tail -> next = P;
    tail = P;
    scanf("%d", &x);
  }
  tail -> next = NULL;
  
}
/*
查找：按位置查找
1. 做边界判断  查找位置不能小于一 如果是0就返回头指针 
2. 做循环 依次赋值出 当前结点的指针
*/
LNode* GetElement(LinkList L, int i) {
  int j = 1;
  if(i == 0) return L;
  if(i < 1) return NULL;

  P = L -> next;
  while (j < i)
  {
    P = P -> next;
    j++;

  return P;
}


/*
向第i个位置插入元素 方法一
*/
bool order_insert_list_1(LinkList &L, int target, ElemType el) {
  LinkList x;
  int length;
  x = L -> next;
  while (x)
  {
    length++;
    x = x -> next;
  }
  
  // 边界判断
  if(target < 1 || target > length+1) return false;
  int j = 1;
  LinkList P = (LinkList)malloc(sizeof(LNode));
  P -> data = el;
  x = L -> next;
  while (x && j < target - 1)
  {
    x = x -> next;
    j++;
  }
  P -> next = x -> next;
  x -> next = P;
  return true;
}

/*
向第i个位置插入元素 方法二 借助GetElement() 获得 i-1元素 的地址
*/
bool order_insert_list_2(LinkList &L, int target, ElemType el) {
  LinkList P = GetElement(L, target - 1);
  if(P == NULL) return false;
  LinkList N = (LinkList)malloc(sizeof(LNode));
  N -> data = el;
  N -> next = P -> next;
  P -> next = N;
  return true;
}
int main(){
  LNode* L; // 链表头指针 结构体指针类型
  tail_insert_list(L); // 尾部插入
  PrintLinkList(L);

  // order_insert_list_1(L, 2, 66);
  order_insert_list_2(L, 2, 66);
  PrintLinkList(L);
  return 0;
}

1.2.5 单链表的删除 #

C
#include <stdio.h>
#include <stdlib.h>

typedef int ElemType;
typedef struct LNode
{
  ElemType data; // 数据域
  struct LNode *next; // 指针域
  
}LNode, *LinkList;


void PrintLinkList(LNode* L) {
  LinkList add = L->next;
  L = L -> next;
  while (L != NULL)
  {
    printf("%3d    %x\n", L->data, add);
    add = L->next;
    L =  L -> next;
  }

  printf("\n");
  
}


/*
尾插法
1. 定义头指针
2. 申请头结点，头指针指向头结点
3. 定义一个尾结点指针 用来指向最后一个结点
4. scanf输入数据 申请结点空间 头结点next指针指向新结点next指针 保存尾结点指针
*/

void tail_insert_list(LinkList &L) {
  L = (LinkList)malloc(sizeof(LNode));
  L -> next = NULL;
  LinkList tail = L, P;
  ElemType x;

  scanf("%d", &x);


  while (x != 9999)
  {
    P = (LinkList)(malloc(sizeof(LNode)));
    P -> data = x;
    tail -> next = P;
    tail = P;
    scanf("%d", &x);
  }
  tail -> next = NULL;
  
}
/*
查找：按位置查找
1. 做边界判断  查找位置不能小于一 如果是0就返回头指针 
2. 做循环 依次赋值出 当前结点的指针
*/
LNode* GetElement(LinkList L, int i) {
  int j = 1;
  if(i == 0) return L;
  if(i < 1) return NULL;
  LinkList P;
  P = L -> next;
  while (j < i)
  {
    P = P -> next;
    j++;
  }
  return P;
}


/*
删除元素 根据 GetElement() 来寻找i-1个元素 和 第二个元素
把第i个元素 释放掉
*/
bool DeleteElement(LinkList &L, int target) {
  LinkList P1 = GetElement(L, target-1);
  LinkList P2 = GetElement(L, target);
  if (P1 == NULL || P2 == NULL) return false;
  P1 -> next = P2 -> next;
  free(P2);
  return true;
}
int main(){
  LNode* L; // 链表头指针 结构体指针类型
  tail_insert_list(L); // 尾部插入
  PrintLinkList(L);
  DeleteElement(L, 2);
  PrintLinkList(L);
  return 0;
}
